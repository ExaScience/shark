99  template<int ndim>
100  inline coords<ndim> coords_range<ndim>::periodic_equiv(coords<ndim> ii) const {
101  seq<0,ndim>::for_each([this,&ii](int d) {
102  coord nd = upper[d] - lower[d];
103  if(ii[d] < lower[d]) {
104  do {
105  ii[d] += nd;
106  } while(ii[d] < lower[d]);
107  } else if(ii[d] >= upper[d]) {
108  do {
109  ii[d] -= nd;
110  } while(ii[d] >= upper[d]);
111  }
112  });
113  return ii;
114  }
