381 template<int ndim,typename T>
382 Future<void> GlobalArray<ndim,T>::iupdate(long k) const {
383  const coords<ndim> gw = ghost_width();
384  typename Domain<ndim>::pcoords np = domain().np;
385  typename Domain<ndim>::pcoords ip = domain().indexp();
386  Access<ndim,T> acc(*this);
387 
388 #if defined(SHARK_MPI_COMM)
389  const bool gc = ghost_corners();
390  MPI_Comm comm = domain().group.impl->comm;
391  unique_ptr<UpdateHandle<ndim>> h(new UpdateHandle<ndim>());
392 #endif
393 
394  for(int di = 0; di < ndim; di++) {
395  if(gw[di] > 0) {
396 #ifndef NDEBUG
397  if(log_mask[verbose_update]) {
398  log_out() << "update back: " << ghost_back[di] << endl;
399  log_out() << "update front: " << ghost_front[di] << endl;
400  }
401 #endif
402  // 1) Consider updating ghosts based on boundary
403  typename Boundary<ndim,T>::general_type* b = dynamic_cast<typename Boundary<ndim,T>::general_type*>(bd[di].t.get());
404  if(b != nullptr) {
405  if(ip[di] == 0)
406  b->set(acc, ghost_back[di], k);
407  if(ip[di] == np[di]-1)
408  b->set(acc, ghost_front[di], k);
409  }
410 
411  // 2) Consider updating ghosts based on communication with neighbors (or periodic equivalents)
412  bool pd = dynamic_cast<typename Boundary<ndim,T>::periodic_type*>(bd[di].t.get()) != nullptr;
413 #if defined(SHARK_MPI_COMM)
414  int prev = domain().shiftd(di, -1, pd);
415  int next = domain().shiftd(di, 1, pd);
416  // backward
417  MPI_Isend(&acc(ghost_back[di].adj(di, 1)), 1, impl->ghost[di], prev, 2*di, comm, &h->req[4*di]);
418  MPI_Irecv(&acc(ghost_front[di].lower), 1, impl->ghost[di], next, 2*di, comm, &h->req[4*di+1]);
419  // forward
420  MPI_Isend(&acc(ghost_front[di].adj(di, -1)), 1, impl->ghost[di], next, 2*di+1, comm, &h->req[4*di+2]);
421  MPI_Irecv(&acc(ghost_back[di].lower), 1, impl->ghost[di], prev, 2*di+1, comm, &h->req[4*di+3]);
422  if(gc)
423  MPI_Waitall(4, &h->req[4*di], MPI_STATUSES_IGNORE);
424 #elif defined(SHARK_NO_COMM)
425  if(pd) {
426  {
427  coords<ndim> off = ghost_back[di].adj(di, 1) - ghost_front[di].lower;
428  ghost_front[di].for_each([&acc,&off](coords<ndim> ii) {
429  acc(ii) = acc(ii+off);
430  });
431  }
432  {
433  coords<ndim> off = ghost_front[di].adj(di, -1) - ghost_back[di].lower;
434  ghost_back[di].for_each([&acc,&off](coords<ndim> ii) {
435  acc(ii) = acc(ii+off);
436  });
437  }
438  }
439 #else
440 #error "No comm update"
441 #endif
442 
443  }
444  }
445 
446 #if defined(SHARK_MPI_COMM)
447  if(!gc)
448  return Future<void>(std::move(h));
449  else
450  return Future<void>(make_unique<DoneHandle>());
451 #elif defined(SHARK_NO_COMM)
452  return Future<void>(make_unique<DoneHandle>());
453 #endif
454 }
