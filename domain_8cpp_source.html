19  vector<int> factor(int n) {
20  vector<int> divs;
21  int k = 2;
22  while(n > 1) {
23  while(n % k > 0)
24  k++;
25  divs.push_back(k);

27  }

31  // This function is a reimplementation of MPI_Dims_create
32  void assign_dims(int n, int ndim, int dims[]) {
33  vector<bool> skip(ndim);
34  for(int d = 0; d < ndim; d++)
35  if(dims[d] != 0) {
36  skip[d] = true;
37  n /= dims[d];
38  } else
39  dims[d] = 1;
40  vector<int> facs = factor(n);
41  int d = -1;
42  for(auto it = facs.crbegin(); it != facs.crend(); ++it) {
43  do {
44  d++;
45  if(d == ndim)
46  d = 0;
47  } while(skip[d]);
48  dims[d] *= *it;
49  }
50  }
51 
52 }
