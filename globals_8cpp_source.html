193 #if defined(SHARK_PTHREAD_SCHED)
194  // Set expected
195  rdy_count = nthrds-1;
196  // Release threads
197  epoch++;
198  // Busy-wait for expected

227 void shark::SetupThreads() {
228 #if defined(SHARK_SER_SCHED)
229  assert(nthrds == 1);
230 #elif defined(SHARK_PTHREAD_SCHED)
231  work = 0;
232  pthread_spin_init(&rdy_count_lock, PTHREAD_PROCESS_PRIVATE);
233  children = new pthread_t[nthrds-1];
234  // Set expected
235  rdy_count = nthrds-1;
236  epoch = 0;
237  // Release threads
238  pthread_attr_t attr;
239  pthread_attr_init(&attr);
240  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
241  for(int k = 0; k < nthrds-1; k++)
242  pthread_create(&children[k], &attr, ThreadMain, reinterpret_cast<void*>(static_cast<size_t>(k+1)));
243  pthread_attr_destroy(&attr);
244  // Busy wait for expected
245  while(rdy_count > 0)
246  ;
247 #elif defined(SHARK_OMP_SCHED)
248  omp_set_num_threads(nthrds);
249 #elif defined(SHARK_TBB_SCHED)
250  tbb_init = new tbb::task_scheduler_init(nthrds);
251 #elif defined(SHARK_COBRA_SCHED)
252  sch = new cobra::scheduler(nthrds);
253 #endif
254 }

284 void* shark::ThreadMain(void* arg) {
285  int threadid = reinterpret_cast<size_t>(arg);
286  int next_epoch = 0;
287  while(true) {
288  pthread_spin_lock(&rdy_count_lock);
289  rdy_count--;
290  pthread_spin_unlock(&rdy_count_lock);
291  // busy wait
292  next_epoch++;
293  while(epoch != next_epoch)
294  ;
295  if(!work)
296  break;
297  work(threadid);
298  }
299  pthread_spin_lock(&rdy_count_lock);
300  rdy_count--;
301  pthread_spin_unlock(&rdy_count_lock);
302  return NULL;
303 }
