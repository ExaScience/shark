132  work = 0;
133  pthread_spin_init(&rdy_count_lock, PTHREAD_PROCESS_PRIVATE);
134  children = new pthread_t[nthrds-1];
135  // Set expected
136  rdy_count = nthrds-1;
137  epoch = 0;
138  // Release threads

142  for(int k = 0; k < nthrds-1; k++)
143  pthread_create(&children[k], &attr, ThreadMain, reinterpret_cast<void*>(static_cast<size_t>(k+1)));
144  pthread_attr_destroy(&attr);

185 void* shark::ThreadMain(void* arg) {
186  int threadid = reinterpret_cast<size_t>(arg);
187  int next_epoch = 0;
188  while(true) {
189  pthread_spin_lock(&rdy_count_lock);
190  rdy_count--;
191  pthread_spin_unlock(&rdy_count_lock);
192  // busy wait
193  next_epoch++;
194  while(epoch != next_epoch)
195  ;
196  if(!work)
197  break;
198  work(threadid);
199  }
200  pthread_spin_lock(&rdy_count_lock);
201  rdy_count--;
202  pthread_spin_unlock(&rdy_count_lock);
203  return NULL;
204 }
