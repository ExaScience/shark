35 template<int ndim, typename T>
36 MPI_Datatype mpi_type_block<ndim,T>::type(coords<ndim> n, array<size_t,ndim-1> eld) {
37  MPI_Datatype t;
38  MPI_Aint lb, extent;
39  if(mpi_type<T>::count() != 1)
40  MPI_Type_contiguous(mpi_type<T>::count(), mpi_type<T>::t, &t);
41  else
42  MPI_Type_dup(mpi_type<T>::t, &t);
43  MPI_Type_get_extent(t, &lb, &extent);
44  for(int di = ndim-1; di >= 0; di--) {
45  MPI_Datatype tmp;
46  tmp = t;
47  MPI_Type_create_hvector(n[di], 1, di == ndim-1 ? extent : eld[di]*extent, tmp, &t);
48  MPI_Type_free(&tmp);
49  }
50  MPI_Type_commit(&t);
51  return t;
52 }
