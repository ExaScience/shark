315  template<int ndim, typename T>
316  inline coords_range<ndim> GlobalArray<ndim,T>::outer_front(int d) const {
317  assert(d >= 0 && d < ndim);
318  const auto gw = ghost_width();
319  auto r = domain().local();
320  for(int i = 0; i<d; ++i) {
321  r.lower[i] += gw[i];
322  r.upper[i] -= gw[i];
323  }
324  r.upper[d] = r.lower[d] + ghost_width()[d];
325  return r;

433  template<int ndim, typename T> template<typename S>
434  GADest<ndim,T>& GADest<ndim,T>::operator=(const S& src) {
435  static_assert(S::number_of_dimensions == ndim, "source dimensionality");
436  assert(ga.domain() == src.domain());
437  Access<ndim,T> d(ga);
438  const typename S::accessor s(src);
439 
440  if (pack) {
441  auto regions = ga.split(region);
442  for(auto r : regions) {
443  assert(src.region().contains(r));
444  std::vector<GABuf<ndim,T> *> ghosts;
445  GABuf<ndim,T> *full_ghost = 0;
446 
447  auto push = [this, &full_ghost, &ghosts,&r](GABuf<ndim,T> *g) {
448  auto overlap = g->r.overlap(r);
449  if (g->r.count() > 0) ga.log_out() << " overlap " << 100 * overlap.count() / g->r.count() << "%" << std::endl;
450  if (!full_ghost && overlap.count() > 0 && overlap.count() == g->r.count()) {
451  full_ghost = g;
452  g->full = true;
453  } else if (overlap.count() > 0) {
454  ghosts.push_back(g);
455  g->full = true;
456  }
457  };
458 
459  seq<0,ndim>::for_each([this, &r, &full_ghost, &ghosts, &push](int d) {
460  push(&ga.g_out_front[d]);
461  push(&ga.g_out_back[d]);
462  });
463 
464  //ga.log_out() << "total ghosts that overlap: " << ghosts.size() << " out of " << regions.size() << std::endl;
465 
466  if(full_ghost && ghosts.size()) {
467  //SHARK_COUNTER("pack_full+ghost");
468  ga.domain().for_each(r, [this, &d, &s, &full_ghost, &ghosts](coords<ndim> i){
469  auto v = d(i) = s(i);
470  full_ghost->da(i) = v;
471  for(auto g : ghosts) if (g->r.contains(i)) g->da(i) = v;
472  });
473  } else if (full_ghost) {
474  //SHARK_COUNTER("pack_onlyfull");
475  ga.domain().for_each(r, [this, &d, &s, &full_ghost](coords<ndim> i){
476  auto v = d(i) = s(i);
477  full_ghost->da(i) = v;
478  });
479  } else if (ghosts.size()) {
480  //SHARK_COUNTER("pack_ghost");
481  ga.domain().for_each(r, [this, &d, &s, &ghosts](coords<ndim> i){
482  auto v = d(i) = s(i);
483  for(auto g : ghosts) if (g->r.contains(i)) g->da(i) = v;
484  });

488  d(i) = s(i);
489  });
490  }

495  d(i) = s(i);
